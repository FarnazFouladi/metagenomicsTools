rm(list=ls())

samplePerPerson <- 20
numPersonsGroupA <- 10
numPersonsGroupB <- 10

data<- vector();
persons <- vector()

meansA <- rnorm(numPersonsGroupA , mean=0)
meansB <- rnorm(numPersonsGroupB , mean=2)
person = 1;

index <-1 
for( i in 1:numPersonsGroupA )
{
	for( j in 1: samplePerPerson)
	{
		data[index] <- rnorm(1, mean=meansA[i])
		persons[index] <- paste("Aperson", person, sep="")
		index = index + 1
	}
	person = person + 1;
}

for( j in 1:numPersonsGroupB )
{
	for( j in 1: samplePerPerson)
	{
		data[index] <- rnorm(1, mean=meansB[i])
		persons[index] <- paste("Bperson", person, sep="")
		index = index + 1
	}
	
	person = person + 1;
}

dFrame <- data.frame(data,persons)
boxplot(data ~ persons)
