setwd("C:\\lactoCheck")

myT <- read.table("closedRefLoggedPlusPCOA_SamplesPlusNegative.txt", sep="\t", header=TRUE)

colors <- vector()

for( i in 1:nrow(myT))
{
	if( myT$groupNumber[i] == 0) 
	{
		colors[i] = "red"			
	}
	else
	{
		if( myT$birthMode[i] == "C" )
			colors[i] = "blue"
		else
			colors[i] = "black"
	}
}

plot(myT$MDS1, myT$MDS2,pch=16,
col=colors,cex=ifelse( myT$stoolOrGa=="G" , 1.25, 2),
xlab="MDS1", ylab="MDS2")
