setwd("G:\\MarkLyteTasteManuscript\\rdpAnalysis")

level <- "genus"
inFileName <- paste( "sparseThreeColumn_", level, "_AsColumnsLogNormalizedPlusMetadata.txt", sep ="")

myT <-read.table(inFileName,header=TRUE,sep="\t")
	numCols <- ncol(myT)
	myColClasses <- c(rep("character",5), rep("numeric", numCols-5))
	myT <-read.table(inFileName,header=TRUE,sep="\t",colClasses=myColClasses)

taxa <- c(
"Olsenella","Blautia","Desulfovibrio","Collinsella","Clostridium.XVIII","Acetitomaculum",
	"Asaccharobacter","Flavonifractor","Papillibacter","Marvinbryantia","Clostridium.IV",
	"Barnesiella","Desulfomicrobium","Anaerovorax","Anaerofustis",
	"Mogibacterium"
)

taxa<- sort(taxa)

par(mfrow=c(4,4))
for( t in taxa ) 
{
	colIndex = which(names(myT) == t)
	
	boxplot( myT[ myT$tissue=="Cecal Content" & myT$line == "High" ,colIndex ],
				myT[ myT$tissue=="Cecal Content" & myT$line == "Low", colIndex ],
					myT[ myT$tissue=="Colon content" & myT$line == "High",colIndex ],
					myT[ myT$tissue=="Colon content" & myT$line == "Low",colIndex ] )
}
