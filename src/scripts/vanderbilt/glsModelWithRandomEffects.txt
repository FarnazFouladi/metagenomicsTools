	myT <-read.table(inFileName,header=TRUE,sep="\t")

	myColClasses <- c(rep("character",8), rep("numeric", numCols-8))
	myT <-read.table(inFileName,header=TRUE,sep="\t",colClasses=myColClasses)
	
	
	pValuesTime <- vector()
	pValuesTreatment <- vector()
	names <- vector()
	
	index <- 1
	for( i in 9:numCols) 
	{

		if( sum(subTable[,i] != 0) >= nrow(myT)/4 )
		{
			bug <- subTable[,i]
			type <- subTable$type
			treatment <- subTable$treatment
			subjectID <- subTable$subjectID
			myFrame <- data.frame( bug,type,treatment, subjectID )
			
			fullModel <- gls( bug~  treatment, method="REML",correlation=corCompSymm(form=~1|factor(subjectID)),data = myFrame )
			reducedModel <- gls( bug~  treatment, method="REML",data = myFrame )

			coefs$p.z <- 2 * (1 - pnorm(abs(coefs$t.value)))
			pValuesTreatment[index] <- coefs[5]$Pr...t..[3];
			pValuesTime[index] <- coefs[5]$Pr...t..[2]
			names[index] = names(subTable)[i]
			index <- index + 1
		}		
	}

	dFrame <- dFrame [order(dFrame$pValuesTreatment),]
