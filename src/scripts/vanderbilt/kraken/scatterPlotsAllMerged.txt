getColors <- function(data) 
{	
	colors <- vector();
	
	for( i in 1:length(data) )
	{
		if( data[i]== "stool" )
			colors[i] <- "red"
		else
			colors[i] <- "blue"
	}
	
	return (colors)
}

getIncludedTaxa <- function(myT)
{
	taxa <- vector();
	data <- myT$maxKrakenFraction
	index <-1
	
	for ( i in 1:nrow(myT))
	{
		if(data[i] > 0.01) 
		{
			taxa[index] <- paste( myT[i,]$taxa, "", sep="")
			index <- index + 1;
		}
	}
	
	taxa <- unique( taxa )
	taxa <- sort(taxa)
	return ( taxa )
	
}

taxa <- getIncludedTaxa(myT)
minVal <- 1

for( i in 1 : length(taxa)) 
{
	aTitle <- paste( taxa[i])
	
	myLm <- lm(log10( myT$krakenLevel[myT$taxa==taxa[i]]+minVal) 
		~ log10(myT$kraken16SLevel[myT$taxa==taxa[i]]+minVal) * myT$isStoolOrSwab[myT$taxa==taxa[i]])
	anova(myLm)
	
	plot( log10( myT$krakenLevel[myT$taxa==taxa[i]]+minVal) , log10(myT$kraken16SLevel[myT$taxa==taxa[i]]+minVal),
				,col =  getColors(myT$isStoolOrSwab[myT$taxa==taxa[i]]) , main=aTitle, xlab="1og10(krakenWholeGenome)",
				ylab="log10(kraken16S)",pch=15,cex=1.3,cex.lab=1.3,cex.main=1.3)

}

write.table(myT, file=paste(level,"_wgsVs16AllSamples.txt", sep =""), row.names=FALSE)
